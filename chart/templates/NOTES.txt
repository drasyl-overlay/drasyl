Get the node endpoint by running these commands:

  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "drasyl.fullname" . }})

Get pod description:

  export POD_NAME=$(kubectl get --namespace default  -o jsonpath="{.items[0].metadata.name}" pods)
  kubectl describe pod $POD_NAME

Get node logs:

  export POD_NAME=$(kubectl get --namespace default  -o jsonpath="{.items[0].metadata.name}" pods)
  kubectl logs -f $POD_NAME
